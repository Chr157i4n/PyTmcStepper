165 def set_current_peak( 

166 self, 

167 run_current: int, 

168 hold_current_multiplier: float = 0.5, 

169 hold_current_delay: int = 10, 

170 ) -> int: 

171 """sets the Peak current for the motor. 

172 

173 Args: 

174 run_current (int): current during movement in mA 

175 hold_current_multiplier (int):current multiplier during standstill (Default value = 0.5) 

176 hold_current_delay (int): delay after standstill after which cur drops (Default value = 10) 

177 run_current_delay (int): delay after movement start after which cur rises (Default value = 0) 

178 rref (int): reference resistor in kOhm (Default value = 12) 

179 

180 Returns: 

181 int: theoretical final current in mA 

182 """ 

183 self.tmc_logger.log(f"Desired peak current: {run_current} mA", Loglevel.DEBUG) 

184 

185 rsense = 0.075 # ohm 

186 vfs = 0.325 # V 

187 

188 current_fs = vfs / (rsense) * 1000 # in mA 

189 

190 self.tmc_logger.log( 

191 f"current_fs: {current_fs:.0f} mA | {current_fs/1000:.1f} A", Loglevel.DEBUG 

192 ) 

193 

194 # 256 == 0 -> max current 

195 global_scaler = round(run_current / current_fs * 256) 

196 

197 global_scaler = min(global_scaler, 256) 

198 global_scaler = max(global_scaler, 0) 

199 

200 self.tmc_logger.log(f"global_scaler: {global_scaler}", Loglevel.DEBUG) 

201 self._set_global_scaler(global_scaler) 

202 

203 ct_current_ma = round(current_fs * global_scaler / 256) 

204 self.tmc_logger.log( 

205 f"Calculated theoretical peak current after gscaler: {ct_current_ma} mA", 

206 Loglevel.DEBUG, 

207 ) 

208 

209 cs_irun = round(run_current / ct_current_ma * 31) 

210 

211 cs_irun = min(cs_irun, 31) 

212 cs_irun = max(cs_irun, 0) 

213 

214 cs_ihold = hold_current_multiplier * cs_irun 

215 

216 cs_irun = round(cs_irun) 

217 cs_ihold = round(cs_ihold) 

218 hold_current_delay = round(hold_current_delay) 

219 

220 self.tmc_logger.log(f"CS_IRun: {cs_irun}", Loglevel.DEBUG) 

221 self.tmc_logger.log(f"CS_IHold: {cs_ihold}", Loglevel.DEBUG) 

222 self.tmc_logger.log(f"IHold_Delay: {hold_current_delay}", Loglevel.DEBUG) 

223 

224 self._set_irun_ihold(cs_ihold, cs_irun, hold_current_delay) 

225 

226 ct_current_ma = round(ct_current_ma * cs_irun / 31) 

227 self.tmc_logger.log( 

228 f"Calculated theoretical final current: {ct_current_ma} mA", Loglevel.INFO 

229 ) 

230 return ct_current_ma
